SELECT CountryID FROM XF.Country
WHERE CountryID = ANY
	(SELECT CountryID FROM XF.City
	GROUP BY CountryID
	HAVING SUM(Population)>400);
	

SELECT CountryID FROM XF.Country
WHERE CountryID = Any
	(SELECT c.CountryID FROM XF.City AS c 
	LEFT JOIN XF.Building AS b ON c.CityID = b.CityID
	WHERE BuildingID IS NULL);
